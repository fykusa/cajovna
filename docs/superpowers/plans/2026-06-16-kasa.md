# Kasa — přehled hotovosti Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prodavačka vidí aktuální stav kasy v POS; admin spravuje pohyby kasy a provádí denní uzávěrku v admin sekci.

**Architecture:** Dvě nové DB tabulky (`91_zaverka`, `90_cashflow`) + nový PHP endpoint `backend/api/kasa.php`. Frontend: nová záložka „Pokladna" v CajovnaPOS (read-only) a nová stránka `/admin/kasa` (plná správa). Stav kasy = poslední confirmed_balance uzávěrky + tržby dnes + pohyby dnes.

**Tech Stack:** PHP 8 / MariaDB (backend), React 19 + TypeScript + Vitest + Testing Library (frontend), CSS Modules (styling).

<?php
require_once __DIR__ . '/../db.php';
require_once __DIR__ . '/../middleware.php';

header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Methods: GET, POST, OPTIONS');
header('Access-Control-Allow-Headers: Content-Type, Authorization');
header('Content-Type: application/json; charset=utf-8');

if ($_SERVER['REQUEST_METHOD'] === 'OPTIONS') {
    http_response_code(204);
    exit;
}

$path   = rtrim(parse_url($_SERVER['REQUEST_URI'], PHP_URL_PATH), '/');
$method = $_SERVER['REQUEST_METHOD'];

if ($method === 'GET' && preg_match('#/api/kasa/status$#', $path)) {
    requireAuth();
    handleStatus();

} elseif ($method === 'POST' && preg_match('#/api/kasa/movements$#', $path)) {
    $auth = requireAdmin();
    handleAddMovement($auth);

} elseif ($method === 'GET' && preg_match('#/api/kasa/movements$#', $path)) {
    requireAdmin();
    handleListMovements();

} elseif ($method === 'POST' && preg_match('#/api/kasa/close$#', $path)) {
    $auth = requireAdmin();
    handleClose($auth);

} elseif ($method === 'GET' && preg_match('#/api/kasa/closings$#', $path)) {
    requireAdmin();
    handleListClosings();

} else {
    http_response_code(404);
    echo json_encode(['error' => 'Not found']);
}

// --- Handlery ---

function handleStatus(): void {
    $pdo   = getPDO();
    $today = date('Y-m-d');

    $stmt = $pdo->prepare(
        'SELECT date, confirmed_balance FROM 91_zaverka WHERE date < ? ORDER BY date DESC LIMIT 1'
    );
    $stmt->execute([$today]);
    $lastClosing = $stmt->fetch(PDO::FETCH_ASSOC) ?: null;

    $stmt = $pdo->prepare(
        'SELECT COALESCE(SUM(total_amount), 0) FROM sales WHERE DATE(created_at) = ?'
    );
    $stmt->execute([$today]);
    $trzbyDnes = (float) $stmt->fetchColumn();

    $stmt = $pdo->prepare(
        'SELECT cm.id, cm.date, cm.amount, cm.note, cm.created_by,
                u.username AS created_by_username, cm.created_at
         FROM 90_cashflow cm
         JOIN users u ON u.id = cm.created_by
         WHERE cm.date = ?
         ORDER BY cm.created_at ASC'
    );
    $stmt->execute([$today]);
    $movements = $stmt->fetchAll(PDO::FETCH_ASSOC);
    $pohybySum = (float) array_sum(array_column($movements, 'amount'));

    $stavKasy = $lastClosing !== null
        ? (float) $lastClosing['confirmed_balance'] + $trzbyDnes + $pohybySum
        : null;

    echo json_encode([
        'last_closing' => $lastClosing
            ? ['date' => $lastClosing['date'], 'confirmed_balance' => (float) $lastClosing['confirmed_balance']]
            : null,
        'trzby_dnes'  => $trzbyDnes,
        'pohyby_dnes' => $pohybySum,
        'stav_kasy'   => $stavKasy,
        'movements'   => $movements,
    ]);
}

function handleAddMovement(array $auth): void {
    $data   = json_decode(file_get_contents('php://input'), true);
    $amount = $data['amount'] ?? null;
    $note   = trim($data['note'] ?? '');

    if ($amount === null || !is_numeric($amount)) {
        http_response_code(400);
        echo json_encode(['error' => 'Chybí nebo neplatná částka']);
        return;
    }
    if ($note === '') {
        http_response_code(400);
        echo json_encode(['error' => 'Poznámka je povinná']);
        return;
    }

    $pdo = getPDO();
    $pdo->prepare(
        'INSERT INTO 90_cashflow (date, amount, note, created_by) VALUES (?, ?, ?, ?)'
    )->execute([date('Y-m-d'), (float) $amount, $note, $auth['user_id']]);

    $id  = (int) $pdo->lastInsertId();
    $row = $pdo->prepare(
        'SELECT cm.id, cm.date, cm.amount, cm.note, cm.created_by,
                u.username AS created_by_username, cm.created_at
         FROM 90_cashflow cm JOIN users u ON u.id = cm.created_by WHERE cm.id = ?'
    );
    $row->execute([$id]);
    http_response_code(201);
    echo json_encode($row->fetch(PDO::FETCH_ASSOC));
}

function handleListMovements(): void {
    $pdo  = getPDO();
    $date = $_GET['date'] ?? date('Y-m-d');
    $stmt = $pdo->prepare(
        'SELECT cm.id, cm.date, cm.amount, cm.note, cm.created_by,
                u.username AS created_by_username, cm.created_at
         FROM 90_cashflow cm JOIN users u ON u.id = cm.created_by
         WHERE cm.date = ? ORDER BY cm.created_at ASC'
    );
    $stmt->execute([$date]);
    echo json_encode($stmt->fetchAll(PDO::FETCH_ASSOC));
}

function handleClose(array $auth): void {
    $data             = json_decode(file_get_contents('php://input'), true);
    $confirmedBalance = $data['confirmed_balance'] ?? null;
    $note             = $data['note'] ?? null;

    if ($confirmedBalance === null || !is_numeric($confirmedBalance)) {
        http_response_code(400);
        echo json_encode(['error' => 'Chybí potvrzený zůstatek']);
        return;
    }

    $pdo   = getPDO();
    $today = date('Y-m-d');

    $stmt = $pdo->prepare(
        'SELECT COALESCE(SUM(total_amount), 0) FROM sales WHERE DATE(created_at) = ?'
    );
    $stmt->execute([$today]);
    $trzbyDnes = (float) $stmt->fetchColumn();

    $stmt = $pdo->prepare(
        'SELECT COALESCE(SUM(amount), 0) FROM 90_cashflow WHERE date = ?'
    );
    $stmt->execute([$today]);
    $pohybySum = (float) $stmt->fetchColumn();

    $stmt = $pdo->prepare(
        'SELECT confirmed_balance FROM 91_zaverka WHERE date < ? ORDER BY date DESC LIMIT 1'
    );
    $stmt->execute([$today]);
    $lastBalance = (float) ($stmt->fetchColumn() ?: 0);

    $calculatedBalance = $lastBalance + $trzbyDnes + $pohybySum;

    $pdo->prepare(
        'INSERT INTO 91_zaverka (date, calculated_balance, confirmed_balance, note, created_by)
         VALUES (?, ?, ?, ?, ?)
         ON DUPLICATE KEY UPDATE
           calculated_balance = VALUES(calculated_balance),
           confirmed_balance  = VALUES(confirmed_balance),
           note               = VALUES(note),
           created_by         = VALUES(created_by),
           updated_at         = CURRENT_TIMESTAMP'
    )->execute([$today, $calculatedBalance, (float) $confirmedBalance, $note, $auth['user_id']]);

    $row = $pdo->prepare(
        'SELECT cc.id, cc.date, cc.calculated_balance, cc.confirmed_balance,
                cc.note, cc.created_by, u.username AS created_by_username,
                cc.created_at, cc.updated_at
         FROM 91_zaverka cc JOIN users u ON u.id = cc.created_by WHERE cc.date = ?'
    );
    $row->execute([$today]);
    echo json_encode($row->fetch(PDO::FETCH_ASSOC));
}

function handleListClosings(): void {
    $pdo    = getPDO();
    $where  = ['1=1'];
    $params = [];

    if (!empty($_GET['from'])) { $where[] = 'cc.date >= ?'; $params[] = $_GET['from']; }
    if (!empty($_GET['to']))   { $where[] = 'cc.date <= ?'; $params[] = $_GET['to']; }

    $stmt = $pdo->prepare(
        'SELECT cc.id, cc.date, cc.calculated_balance, cc.confirmed_balance,
                cc.note, cc.created_by, u.username AS created_by_username,
                cc.created_at, cc.updated_at
         FROM 91_zaverka cc JOIN users u ON u.id = cc.created_by
         WHERE ' . implode(' AND ', $where) . '
         ORDER BY cc.date DESC LIMIT 100'
    );
    $stmt->execute($params);
    echo json_encode($stmt->fetchAll(PDO::FETCH_ASSOC));
}

import { apiFetch } from './client'
import type { KasaStatus, CashMovement, CashClosing } from '../types'

export const getKasaStatus = (): Promise<KasaStatus> =>
  apiFetch<KasaStatus>('/kasa/status')

export const addKasaMovement = (amount: number, note: string): Promise<CashMovement> =>
  apiFetch<CashMovement>('/kasa/movements', {
    method: 'POST',
    body: JSON.stringify({ amount, note }),
  })

export const getKasaMovements = (date?: string): Promise<CashMovement[]> =>
  apiFetch<CashMovement[]>(`/kasa/movements${date ? `?date=${date}` : ''}`)

export const closeKasa = (confirmed_balance: number, note?: string): Promise<CashClosing> =>
  apiFetch<CashClosing>('/kasa/close', {
    method: 'POST',
    body: JSON.stringify({ confirmed_balance, note: note ?? null }),
  })

export const getKasaClosings = (from?: string, to?: string): Promise<CashClosing[]> => {
  const parts: string[] = []
  if (from) parts.push(`from=${from}`)
  if (to)   parts.push(`to=${to}`)
  return apiFetch<CashClosing[]>(`/kasa/closings${parts.length ? `?${parts.join('&')}` : ''}`)
}

import { useState, useEffect } from 'react'
import type { KasaStatus, CashMovement, CashClosing } from '../../types'
import { getKasaStatus, addKasaMovement, closeKasa, getKasaClosings } from '../../api/kasa'
import { useToast } from '../../components/toast/useToast'
import { periodRange } from './periodRange'
import styles from './Kasa.module.css'

const fmtKc = (n: number) => Math.round(n).toLocaleString('cs-CZ') + ' Kč'
const fmtDate = (d: string) => {
  const [y, m, day] = d.split('-')
  return `${day}.${m}.${y}`
}

export default function AdminKasa() {
  const toast = useToast()
  const [status, setStatus]           = useState<KasaStatus | null>(null)
  const [statusLoading, setStatusLoading] = useState(true)
  const [showMovForm, setShowMovForm] = useState(false)
  const [movAmount, setMovAmount]     = useState('')
  const [movNote, setMovNote]         = useState('')
  const [movSubmitting, setMovSubmitting] = useState(false)
  const [closeAmount, setCloseAmount] = useState('')
  const [closeNote, setCloseNote]     = useState('')
  const [closeSubmitting, setCloseSubmitting] = useState(false)
  const initRange = periodRange('month')
  const [histFrom, setHistFrom]       = useState(initRange.from)
  const [histTo, setHistTo]           = useState(initRange.to)
  const [closings, setClosings]       = useState<CashClosing[]>([])
  const [histLoading, setHistLoading] = useState(true)

  async function loadStatus() {
    setStatusLoading(true)
    try {
      const s = await getKasaStatus()
      setStatus(s)
      setCloseAmount(String(Math.round(s.stav_kasy ?? (s.trzby_dnes + s.pohyby_dnes))))
    } catch (e) {
      toast.error(e instanceof Error ? e.message : 'Chyba načítání stavu')
    } finally {
      setStatusLoading(false)
    }
  }

  async function loadHistory(from = histFrom, to = histTo) {
    setHistLoading(true)
    try {
      setClosings(await getKasaClosings(from, to))
    } catch (e) {
      toast.error(e instanceof Error ? e.message : 'Chyba načítání historie')
    } finally {
      setHistLoading(false)
    }
  }

  useEffect(() => {
    loadStatus()
    loadHistory()
  }, [])

  async function handleAddMovement(e: React.FormEvent) {
    e.preventDefault()
    const amount = parseFloat(movAmount.replace(',', '.'))
    if (isNaN(amount)) { toast.error('Neplatná částka'); return }
    if (!movNote.trim()) { toast.error('Poznámka je povinná'); return }
    setMovSubmitting(true)
    try {
      await addKasaMovement(amount, movNote.trim())
      toast.success('Pohyb přidán')
      setShowMovForm(false)
      setMovAmount('')
      setMovNote('')
      loadStatus()
    } catch (e) {
      toast.error(e instanceof Error ? e.message : 'Chyba přidání pohybu')
    } finally {
      setMovSubmitting(false)
    }
  }

  async function handleClose(e: React.FormEvent) {
    e.preventDefault()
    const amount = parseFloat(closeAmount.replace(',', '.'))
    if (isNaN(amount)) { toast.error('Neplatný zůstatek'); return }
    setCloseSubmitting(true)
    try {
      await closeKasa(amount, closeNote.trim() || undefined)
      toast.success('Den uzavřen')
      setCloseNote('')
      loadStatus()
      loadHistory()
    } catch (e) {
      toast.error(e instanceof Error ? e.message : 'Chyba uzavření')
    } finally {
      setCloseSubmitting(false)
    }
  }

  const movements: CashMovement[] = status?.movements ?? []

  return (
    <div className={styles.page}>
      <h1 className={styles.title}>Kasa</h1>

      {/* 1. Dnešní stav */}
      <div className={styles.statusGrid}>
        <div className={styles.statCard}>
          <div className={styles.statLabel}>Uzávěrka předchozí den</div>
          <div className={styles.statValue}>
            {statusLoading ? '…' : status?.last_closing ? fmtKc(status.last_closing.confirmed_balance) : '—'}
          </div>
        </div>
        <div className={styles.statCard}>
          <div className={styles.statLabel}>Tržby dnes</div>
          <div className={styles.statValue}>
            {statusLoading ? '…' : status ? fmtKc(status.trzby_dnes) : '—'}
          </div>
        </div>
        <div className={styles.statCard}>
          <div className={styles.statLabel}>Pohyby dnes</div>
          <div className={styles.statValue}>
            {statusLoading ? '…' : status ? fmtKc(status.pohyby_dnes) : '—'}
          </div>
        </div>
        <div className={`${styles.statCard} ${styles.total}`}>
          <div className={styles.statLabel}>Aktuální stav kasy</div>
          <div className={styles.statValue}>
            {statusLoading ? '…'
              : status?.stav_kasy !== null && status?.stav_kasy !== undefined
                ? fmtKc(status.stav_kasy)
                : status ? `— + ${fmtKc(status.trzby_dnes + status.pohyby_dnes)}` : '—'}
          </div>
        </div>
      </div>

      {/* 2. Pohyby dnešního dne */}
      <div className={styles.section}>
        <div className={styles.sectionTitle}>Pohyby dnes</div>
        <table className={styles.table}>
          <thead>
            <tr>
              <th>Čas</th>
              <th>Částka</th>
              <th>Poznámka</th>
              <th>Zapsal</th>
            </tr>
          </thead>
          <tbody>
            {movements.length === 0 ? (
              <tr><td colSpan={4} className={styles.emptyRow}>Žádné pohyby.</td></tr>
            ) : movements.map((m) => (
              <tr key={m.id}>
                <td>{new Date(m.created_at).toLocaleTimeString('cs-CZ', { hour: '2-digit', minute: '2-digit' })}</td>
                <td className={m.amount >= 0 ? styles.amountPos : styles.amountNeg}>
                  {m.amount >= 0 ? '+' : ''}{fmtKc(m.amount)}
                </td>
                <td>{m.note}</td>
                <td>{m.created_by_username}</td>
              </tr>
            ))}
          </tbody>
        </table>

        {!showMovForm && (
          <button className={styles.addBtn} onClick={() => setShowMovForm(true)}>
            + Přidat pohyb
          </button>
        )}

        {showMovForm && (
          <form className={styles.movForm} onSubmit={handleAddMovement}>
            <input
              className={styles.movInput}
              type="number"
              step="0.01"
              placeholder="Částka (záporná = výdej)"
              value={movAmount}
              onChange={(e) => setMovAmount(e.target.value)}
              required
              autoFocus
            />
            <input
              className={`${styles.movInput} ${styles.wide}`}
              type="text"
              placeholder="Poznámka (povinná)"
              value={movNote}
              onChange={(e) => setMovNote(e.target.value)}
              required
            />
            <button className={styles.submitBtn} type="submit" disabled={movSubmitting}>
              {movSubmitting ? 'Ukládám…' : 'Přidat'}
            </button>
            <button className={styles.cancelBtn} type="button" onClick={() => { setShowMovForm(false); setMovAmount(''); setMovNote('') }}>
              Zrušit
            </button>
          </form>
        )}
      </div>

      {/* 3. Uzavření dne */}
      <div className={styles.section}>
        <div className={styles.sectionTitle}>Uzavření dne</div>
        <div className={styles.closeHint}>
          Vypočítaný zůstatek je předvyplněn. Opravte pokud se liší od fyzicky přepočítané kasičky.
        </div>
        <form className={styles.closeForm} onSubmit={handleClose}>
          <input
            className={styles.movInput}
            type="number"
            step="0.01"
            placeholder="Potvrzený zůstatek (Kč)"
            value={closeAmount}
            onChange={(e) => setCloseAmount(e.target.value)}
            required
          />
          <input
            className={`${styles.movInput} ${styles.wide}`}
            type="text"
            placeholder="Poznámka (volitelná)"
            value={closeNote}
            onChange={(e) => setCloseNote(e.target.value)}
          />
          <button className={styles.closeBtn} type="submit" disabled={closeSubmitting}>
            {closeSubmitting ? 'Ukládám…' : 'Uzavřít den'}
          </button>
        </form>
      </div>

      {/* 4. Historie uzávěrek */}
      <div className={styles.section}>
        <div className={styles.sectionTitle}>Historie uzávěrek</div>
        <div className={styles.histFilter}>
          <input
            className={styles.dateInput}
            type="date"
            value={histFrom}
            onChange={(e) => setHistFrom(e.target.value)}
          />
          <span style={{ color: '#888' }}>–</span>
          <input
            className={styles.dateInput}
            type="date"
            value={histTo}
            onChange={(e) => setHistTo(e.target.value)}
          />
          <button
            className={styles.filterBtn}
            onClick={() => loadHistory()}
          >
            Zobrazit

        {histLoading ? (
          <p className={styles.loading}>Načítám…</p>

          <table className={styles.table}>
            <thead>
              <tr>
                <th>Datum</th>
                <th>Vypočítaný</th>
                <th>Potvrzený</th>
                <th>Poznámka</th>
                <th>Uzavřel</th>

            <tbody>
